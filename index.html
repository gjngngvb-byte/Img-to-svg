function renderHatching(ctx, img, w, h, angle, spacing, mode){ const temp = document.createElement("canvas"); temp.width = w; temp.height = h; const tctx = temp.getContext("2d"); tctx.drawImage(img, 0, 0); const imgData = tctx.getImageData(0,0,w,h).data; const rad = angle * Math.PI / 180; const dx = Math.cos(rad); const dy = Math.sin(rad); // passo base controla densidade global const baseStep = spacing; for(let y = 0; y < h; y += 2){ for(let x = 0; x < w; x += 2){ const i = (y*w + x) * 4; const r = imgData[i]; const g = imgData[i+1]; const b = imgData[i+2]; // luminância perceptual (mais realista que média simples) const brightness = 0.299*r + 0.587*g + 0.114*b; // 0 = preto, 1 = branco const intensity = brightness / 255; // 🔥 REGRA CENTRAL: // quanto mais escuro → mais linhas const density = 1 - intensity; // ⚪ branco → nada if(density < 0.08) continue; // calcula espaçamento dinâmico // preto → pequeno espaçamento (muitas linhas) const step = baseStep * (1 - density) + 2; // adiciona jitter para naturalidade if(Math.random() > density) continue; const len = density * 18; const x2 = x + dx * len; const y2 = y + dy * len; ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x2, y2); ctx.stroke(); } } // 🔁 cross-hatching opcional (reforço de sombra) if(mode === "cross"){ const rad2 = (angle + 90) * Math.PI / 180; const dx2 = Math.cos(rad2); const dy2 = Math.sin(rad2); for(let y = 0; y < h; y += 3){ for(let x = 0; x < w; x += 3){ const i = (y*w + x) * 4; const r = imgData[i]; const g = imgData[i+1]; const b = imgData[i+2]; const brightness = 0.299*r + 0.587*g + 0.114*b; const density = 1 - (brightness / 255); if(density < 0.25) continue; const len = density * 14; if(Math.random() > density) continue; ctx.beginPath(); ctx.moveTo(x, y); ctx.lineTo(x + dx2*len, y + dy2*len); ctx.stroke(); } } } }
